PROGRAM
from pymongo import MongoClient
import pandas as pd

# Connect MongoDB

client = MongoClient("mongodb+srv://username:password@cluster0.mongodb.net/")

# Create Database

db = client["StudentDB"]

# Create Collection

collection = db["students"]

# Insert Data

data = {
    "name": "Rahul",
    "age": 21,
    "city": "Mumbai"
}

collection.insert_one(data)

print("Data Inserted Successfully")

# Display Data

records = collection.find()

for record in records:
    print(record)

# Update Data

collection.update_one(
    {"name": "Rahul"},
    {"$set": {"age": 22}}
)

print("Data Updated")

# Delete Data

collection.delete_one({"name": "Rahul"})

print("Data Deleted")
